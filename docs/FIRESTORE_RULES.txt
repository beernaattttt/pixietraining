// Paste this into Firebase Console -> Firestore Database -> Rules
//
// Core idea: the browser and Roblox NEVER talk to Firestore directly.
// Only the Vercel backend (Firebase Admin SDK, service account) touches
// this database. Admin SDK bypasses these rules entirely by design —
// these rules exist purely as a defense-in-depth backstop in case a
// client SDK key ever leaks or someone misconfigures something. Default:
// deny everything from any client.

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    match /{document=**} {
      allow read: if false;
      allow write: if false;
    }

  }
}
